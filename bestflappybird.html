var main = createSprite(250,200,100,100); main.shapeColor = "lightyellow"; var stop = createSprite(260,360); stop.setAnimation("pause_1"); stop.scale = 0.1; var spinner1 = createSprite(80, 100); spinner1.setAnimation("fidget_spinner_1"); spinner1.scale = 0.4; var spinner2 = createSprite(80, 200); spinner2.setAnimation("fidget_spinner_2"); spinner2.scale = 0.4; var spinner3 = createSprite(80, 320); spinner3.setAnimation("fidget_spinner_3"); spinner3.scale = 0.4; function draw() { background("lightyellow"); if(mousePressedOver(spinner1)){ main.setAnimation("fidget_spinner_1"); main.scale = 0.5; } else if((mousePressedOver(spinner2))){ main.setAnimation("fidget_spinner_2"); main.scale = 0.5; } else if((mousePressedOver(spinner3))){ main.setAnimation("fidget_spinner_3"); main.scale = 0.5; } if(mousePressedOver(main)){ //This is where you can change the speed. main.rotationSpeed = main.rotationSpeed + 10; console.log("speed " + main.rotationSpeed); } if(mousePressedOver(stop)){ main.rotationSpeed = 0; } drawSprites(); textSize(22); fill("black"); text("Choose Spinner And Spin Using MousePress", 10, 30); }
